# 1. Встановити залежності
npm install

# 2. Запустити dev сервер
npm run dev

# 3. Запустити тести
npm run test

# 4. Запустити тести з UI
npm run test:ui

# 5. Запустити тести з coverage
npm run test:coverage

# 6. Запустити linter
npm run lint

# 7. Авто-виправлення lint помилок
npm run lint:fix

# 8. Format код
npm run format

# 9. Перевірити format (для CI)
npm run format:check

# 10. Повна валідація
npm run validate

# 11. Зібрати production build
npm run build

# 12. Preview production build
npm run preview

# 13. Analyze bundle size
npm run analyze

# 14. Очистити кеш
npm run clean
